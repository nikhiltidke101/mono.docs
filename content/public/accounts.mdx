## accounts-create

Creates an account, which is a record representing a customer or an
organization.


### Request Body (Optional)



<Property required={false} deprecated={false} name={"description"} type={"string"}>

Description of the account.

</Property>

<Property required={true} deprecated={false} name={"display_name"} type={"string"}>

Name of the account.

</Property>

<Property required={false} deprecated={false} name={"domains"} type={"array of string"}>

List of company's domain names.

<span>Example:  `"companyname.com"`</span>

</Property>

<Property required={false} deprecated={false} name={"external_refs"} type={"array of string"}>

External refs are unique identifiers from your customer system of
records, stored as a list.


</Property>

<Property required={false} deprecated={false} name={"owned_by"} type={"array of string"}>

List of Dev users owning this account.

</Property>

<Property required={false} deprecated={false} name={"tier"} type={"string"}>

The tier of the account.

</Property>

<Property required={false} deprecated={false} name={"websites"} type={"array of string"}>

List of company websites. 


<span>Example:  `["www.companyname.com","www.marketplace.companyname.com"]`</span>

</Property>

| Status code | Description |
| ----------- | ----------- |
| `201` | Success. |
| `400` | 400 Bad Request. |
| `401` | 401 Unauthorized. |
| `403` | 403 Forbidden. |
| `409` | 409 Conflict. |
| `429` | 429 Too Many Requests. |
| `500` | 500 Internal Server Error. |
| `503` | 503 Service Unavailable. |

</APIInfo>

```bash title="curl"
curl -X POST "https://api.devrev.ai/accounts.create" \

  "description": "string",

    "string"

  "owned_by": [
    "string"
  ],
  "tier": "string",
  "websites": [
    "www.companyname.com",
    "www.marketplace.companyname.com"
  ]
}'
```

export type Account = OrgBase & {
  /**
   * Description of the corresponding Account.
   */

   * External refs are unique identifiers from your customer system of

   * List of Dev user IDs owning this Account.
   */

  /**
   * The Tier of the corresponding Account.
   */
  tier?: string;

  /**
   * Name of the Organization.
   */
  display_name?: string;
};
export type UserSummary = DevUserSummary | RevUserSummary | SysUserSummary;
export type DevUserSummary = UserBaseSummary;
export type UserBaseSummary = AtomBaseSummary & {
  /**
   * The user's display name. The name is non-unique and mutable.
   *
   */
  display_name?: string;
  display_picture?: ArtifactSummary;
  /**
   * Email address of the user.
   */
  email?: string;
  /**
   * Full name of the user.
   */
  full_name?: string;
  /**
   * State of the user.
   */
  state?: "active" | "deactivated" | "deleted" | "locked" | "shadow" | "unassigned";
};
export type ArtifactSummary = AtomBaseSummary;
export type RevUserSummary = UserBaseSummary & {
  /**
   * External ref is a mutable unique identifier for a user within the
   * Rev organization from your primary customer record. If none is
   * available, a good alternative is the email address/phone number
   * which could uniquely identify the user. If none is specified, a
   * system-generated identifier will be assigned to the user.
   *
   */
  external_ref?: string;
  rev_org?: OrgSummary;
};
export type OrgSummary = RevOrgSummary;
export type RevOrgSummary = OrgBaseSummary;
export type OrgBaseSummary = AtomBaseSummary & {
  /**
   * Name of the Organization.
   */

};
export type SysUserSummary = UserBaseSummary;

<span>Format:  `"date-time"`</span>
