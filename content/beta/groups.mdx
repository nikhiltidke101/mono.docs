## groups-create

Creates a new group. A group is a collection of users.

### Request Body (Optional)



<Property required={true} deprecated={false} name={"description"} type={"string"}>

Description of the group.

</Property>

<Property required={false} deprecated={false} name={"dynamic_group_info"} type={"object"}>

Information to define dynamic groups.

<Accordions>

<Accordion title={"Object Type"}>



</Accordion>

</Accordions>

</Property>

<Property required={false} deprecated={false} name={"member_type"} type={"string"}>

Type of the members in the group.

<span>Value in:  `"dev_user" | "rev_user"`</span>

</Property>

<Property required={true} deprecated={false} name={"name"} type={"string"}>

Unique name of the group.

</Property>

<Property required={false} deprecated={false} name={"owner"} type={"string"}>

Owner of the group.

</Property>

<Property required={false} deprecated={false} name={"type"} type={"string"}>

Type of the group.

<span>Value in:  `"dynamic" | "static"`</span>

</Property>

| Status code | Description |
| ----------- | ----------- |
| `201` | Success. |
| `400` | 400 Bad Request. |
| `401` | 401 Unauthorized. |
| `403` | 403 Forbidden. |
| `429` | 429 Too Many Requests. |
| `500` | 500 Internal Server Error. |
| `503` | 503 Service Unavailable. |

</APIInfo>

curl -X POST "https://api.devrev.ai/groups.create" \

  "description": "string",
  "dynamic_group_info": {},
  "member_type": "dev_user",

  "owner": "string",

export type Group = AtomBase & {
  /**
   * Description of the group.
   */
  description?: string;
  /**
   * Whether it is a default group.
   */
  is_default: boolean;
  /**
   * Type of the members in the group.
   */
  member_type?: "dev_user" | "rev_user";
  /**
   * Name of the group.
   */
  name?: string;
  owner?: UserSummary;
};
export type UserSummary = DevUserSummary | RevUserSummary | ServiceAccountSummary | SysUserSummary;

   * The user's display name. The name is non-unique and mutable.

   */
  email?: string;

   * Full name of the user.

   */

};
export type ArtifactSummary = AtomBaseSummary;
export type RevUserSummary = UserBaseSummary & {
  /**
   * External ref is a mutable unique identifier for a user within the
   * Rev organization from your primary customer record. If none is
   * available, a good alternative is the email address/phone number
   * which could uniquely identify the user. If none is specified, a
   * system-generated identifier will be assigned to the user.
   *
   */

  rev_org?: OrgSummary;
};
